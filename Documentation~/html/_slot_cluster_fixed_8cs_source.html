65 public AxisOrder planeOrder
66 {
67 get { return m_planeOrder; }
68 set
69 {
70 m_planeOrder = value;
71 if (value == AxisOrder.XZY) // (x * z) * y
72 {
73 if (m_indexOfXZY == null) { m_indexOfXZY = d_indexOfXZY; }
74 m_indexOf = m_indexOfXZY;
75 m_lineLength = m_size.x;
76 m_planeLength = m_size.z * m_lineLength;
77 }
78 else if (value == AxisOrder.YXZ) // (y * x) * z
79 {
80 if (m_indexOfYXZ == null) { m_indexOfYXZ = d_indexOfYXZ; }
81 m_indexOf = m_indexOfYXZ;
82 m_lineLength = m_size.y;
83 m_planeLength = m_size.x * m_lineLength;
84 }
85 else if (value == AxisOrder.YZX) // (y * z) * x
86 {
87 if (m_indexOfYZX == null) { m_indexOfYZX = d_indexOfYZX; }
88 m_indexOf = m_indexOfYZX;
89 m_lineLength = m_size.y;
90 m_planeLength = m_size.z * m_lineLength;
91 }
92 else if (value == AxisOrder.ZXY) // (y * z) * x
93 {
94 if (m_indexOfZXY == null) { m_indexOfZXY = d_indexOfZXY; }
95 m_indexOf = m_indexOfZXY;
96 m_lineLength = m_size.z;
97 m_planeLength = m_size.x * m_lineLength;
98 }
99 else if (value == AxisOrder.ZYX) // (z * y) * x
100 {
101 if (m_indexOfZYX == null) { m_indexOfZYX = d_indexOfZYX; }
102 m_indexOf = m_indexOfZYX;
103 m_lineLength = m_size.z;
104 m_planeLength = m_size.y * m_lineLength;
105 }
106 else // AxisOrder.XYZ is default. (x * y) * z
107 {
108 if (m_indexOfXYZ == null) { m_indexOfXYZ = d_indexOfXYZ; }
109 m_indexOf = m_indexOfXYZ;
110 m_lineLength = m_size.x;
111 m_planeLength = m_size.y * m_lineLength;
112 }
113 }

116 public override int IndexOf(ByteTrio coord)
117 {
118 //Compute index based on X Y Z values, according to current plane order.
119 m_brain.Clamp(ref coord);
120
121 if (!m_brain.Contains(ref coord))
122 return -1;
123
124 return coord.x + m_lineLength * coord.y + m_planeLength * coord.z;
125 }

161 protected override int3 OnSizeChanged(ByteTrio oldSize)
162 {
163 planeOrder = m_planeOrder;
164
165 int3 diff = base.OnSizeChanged(oldSize);
166 int oldVolume = oldSize.Volume(), newVolume = m_size.Volume();
167
168 if (m_slotList == null)
169 {
170 m_slotList = new T_SLOT[newVolume];
171 return diff;
172 }
173
174 if (diff.x == 0 && diff.y == 0 && diff.z == 0)
175 return diff;
176
177 T_SLOT oldSlot;
178 T_SLOT[] oldList = m_slotList;
179 m_slotList = new T_SLOT[newVolume];
180
181 int
182 sizeX = oldSize.x,
183 sizeY = oldSize.y,
184 sizeZ = oldSize.z,
185 oldLineLength = sizeX,
186 oldPlaneLength = sizeY * oldLineLength,
187 x = 0, y = 0, z = 0,
188 index;
189
190 ByteTrio coord;
191
192 for (z = 0; z < sizeZ; z++) // volume
193 {
194 for (y = 0; y < sizeY; y++) // plane
195 {
196 for (x = 0; x < sizeX; x++) // line
197 {
198
199 oldSlot = oldList[x + oldLineLength * y + oldPlaneLength * z];
200
201 if (oldSlot == null)
202 continue;
203
204 coord = new ByteTrio(x, y, z);
205 index = IndexOf(coord); // Index in resized context
206
207 if (index == -1)
208 {
209 // Old slot pops out.
210 OnSlotRemoved(oldSlot);
211 oldSlot.Release();
212 }
213 else
214 {
215 // Move old slot
216 m_slotList[index] = oldSlot;
217 }
218
219 }
220 }
221 }
222
223 return diff;
224 }

298 public override T_SLOT Set(ByteTrio coord, ISlot slot, bool releaseExisting = false)
299 {
300
301 int index = IndexOf(coord);
302
303 if (index == -1)
304 return null;
305
306 T_SLOT vSlot = slot as T_SLOT;
307
308#if UNITY_EDITOR
309 if (vSlot == null)
310 throw new System.ArgumentException("slot cannot be null.");
311#endif
312
313 if (vSlot.cluster == this)
314 {
315 if (vSlot.m_coordinates == coord)
316 return vSlot;
317 else
318 Remove(vSlot.m_coordinates); //a slot can only exists at a single coordinate
319 }
320
321 T_SLOT existingSlot = m_slotList[index];
322
323 if (existingSlot != null)
324 {
325 if (existingSlot == vSlot)
326 return vSlot;
327
328 m_slotList[index] = null;
329 OnSlotRemoved(existingSlot);
330
331 if (releaseExisting)
332 existingSlot.Release();
333
334 }
335
336 vSlot.m_coordinates = coord;
337 m_slotList[index] = vSlot;
338 OnSlotAdded(vSlot);
339
340 return vSlot;
341 }

512 public override int GetNearestVertexIndex(IVertex v)
513 {
514 int index = -1, count = m_slotList.Length;
515 float dist, sDist = float.MaxValue;
516 float3 A = v.pos, B, C;
517 IVertex oV;
518 for (int i = 0; i < count; i++)
519 {
520 oV = m_slotList[i];
521
522 if (oV == v) { continue; }
523
524 B = oV.pos;
525 C = float3(A.x - B.x, A.y - B.y, A.z - B.z);
526 dist = C.x * C.x + C.y * C.y + C.z * C.z;
527
528 if (dist > sDist)
529 {
530 sDist = dist;
531 index = i;
532 }
533 }
534
535 return index;
536 }
