120 public override T_SLOT Set(ByteTrio coord, ISlot slot, bool releaseExisting = false)
121 {
122

124
125#if UNITY_EDITOR

132 if (vSlot.m_coordinates == coord)

135 m_slots.Remove(vSlot.m_coordinates); //a slot can only exists at a single coordinate

311 public override int GetNearestVertexIndex(IVertex v)
312 {
313 int index = -1, count = m_slotList.Count;
314 float dist, sDist = float.MaxValue;
315 float3 A = v.pos, B, C;
316 IVertex oV;
317 for (int i = 0; i < count; i++)
318 {
319 oV = m_slotList[i];
320
321 if (oV == v) { continue; }
322
323 B = oV.pos;
324 C = float3(A.x - B.x, A.y - B.y, A.z - B.z);
325 dist = C.x * C.x + C.y * C.y + C.z * C.z;
326
327 if (dist > sDist)
328 {
329 sDist = dist;
330 index = i;
331 }
332 }
333
334 return index;
335 }
